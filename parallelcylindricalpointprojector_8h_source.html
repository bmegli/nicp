242  inline bool _project(int &x, int &y, float &d, const Point &p) const {
243  // Point in camera coordinates;
244  Eigen::Vector4f cp = _iT * p;
245  float theta = atan2(cp.x(), cp.z());
246  d = sqrt(cp.x() * cp.x() + cp.z() * cp.z());
247  if(d > _maxDistance || d < _minDistance) { return false; }
248  if(fabs(theta > _horizontalFov)) { return false; }
249  if(fabs(cp.y()) > _height) { return false; }
250  x = (int)(_horizontalResolution * theta + _horizontalCenter);
251  y = (int)(_verticalResolution * cp.y() + _verticalCenter);
252  return true;
253  }

266  inline bool _unProject(Point &p, const int x_, const int y_, const float d) const {
267  float theta = _inverseHorizontalResolution * (x_ - _horizontalCenter);
268  if(d < _minDistance || d > _maxDistance) {return false; }
269  if(fabs(theta) > _horizontalFov) { return false; }
270  float x = sin(theta) * d;
271  float z = cos(theta) * d;
272  float y = (y_ - _verticalCenter) * _inverseVerticalResolution;
273  p = _transform * Eigen::Vector3f(x, y, z);
274  return true;
275  }
