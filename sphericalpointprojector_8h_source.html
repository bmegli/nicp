252  inline bool _project(int &x, int &y, float &d, const Point &p) const {
253  // Point in camera coordinates;
254  Eigen::Vector4f cp = _iT * p;
255  float theta = atan2(cp.x(), cp.z());
256  d = sqrt(cp.x() * cp.x() + cp.z() * cp.z());
257  float phi = atan2(cp.y(), d);
258  if(d > _maxDistance || d < _minDistance) { return false; }
259  if(fabs(theta > _horizontalFov)) { return false; }
260  if(fabs(phi > _verticalFov)) { return false; }
261  x = (int)(_horizontalResolution * theta + _horizontalCenter);
262  y = (int)(_verticalResolution * phi + _verticalCenter);
263  return true;
264  }

277  inline bool _unProject(Point &p, const int x_, const int y_, const float d) const {
278  if(d < _minDistance || d > _maxDistance)
279  return false;
280  float theta = _inverseHorizontalResolution * (x_ - _horizontalCenter);
281  float phi = _inverseVerticalResolution * (y_ - _verticalCenter);
282  if(fabs(theta) > _horizontalFov) { return false; }
283  if(fabs(phi) > _verticalFov) { return false; }
284  float x = sin(theta) * d;
285  float z = cos(theta) * d;
286  float y = d * tanf(phi);
287  p = _transform * Eigen::Vector3f(x, y, z);
288  return true;
289  }
